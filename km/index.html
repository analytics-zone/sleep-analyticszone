នាំចេញ និងវិភាគទិន្នន័យ Apple Health របស់អ្នក

បំប្លែងទិន្នន័យ HealthKit ឆៅទៅជាការយល់ដឹងដែលអាចអនុវត្តបាន និងរបាយការណ៍ដែលអាចចែករំលែកបាន។ នាំចេញជា CSV, JSON និង PDF។ ចែករំលែកជាមួយគ្រូពេទ្យ។ តាមដានជីវសញ្ញាសម្រាប់ភាពយូរអង្វែង។ រក្សាឯកជនភាព ១០០%។ ផ្អែកលើវិទ្យាសាស្ត្រ។

ទាញយកឥតគិតថ្លៃមើលវិទ្យាសាស្ត្រ ↓

វិទ្យាសាស្ត្រនៃសុខភាពរបស់អ្នក

យើងមិនត្រឹមតែបង្ហាញទិន្នន័យប៉ុណ្ណោះទេ។ យើងបកស្រាយជីវមាត្ររបស់អ្នកដោយផ្អែកលើការស្រាវជ្រាវគ្លីនិក និងការសិក្សាដែលបានត្រួតពិនិត្យដោយអ្នកជំនាញ។

សុខភាពបេះដូង និងសួត (VO₂ Max)

ជារឿយៗត្រូវបានចាត់ទុកថាជាកត្តាកំណត់អាយុជីវិតដ៏ខ្លាំងក្លាបំផុត។ ការវិភាគរបស់យើងវាយតម្លៃ VO₂ Max ប៉ាន់ស្មានរបស់អ្នកធៀបនឹងបទដ្ឋានប្រជាជន ដើម្បីវាយតម្លៃអាយុជីវសាស្ត្រ និងហានិភ័យមរណភាពរយៈពេលវែងរបស់អ្នក។

ហេតុអ្វីបានជាវាសំខាន់៖ ការសិក្សាដ៏សំខាន់មួយលើអ្នកជំងឺ ១២២,០០៧ នាក់បានរកឃើញថា សុខភាពបេះដូង និងសួតកម្រិត \"elite\" ត្រូវបានផ្សារភ្ជាប់ជាមួយនឹងការកាត់បន្ថយហានិភ័យមរណភាព ៨០% ធៀបនឹងសុខភាពខ្សោយ។

ប្រភព៖ Mandsager K, et al. (JAMA Network Open, 2018).

តុល្យភាពប្រព័ន្ធប្រសាទស្វ័យប្រវត្តិ (HRV)

យើងវិភាគភាពប្រែប្រួលនៃចង្វាក់បេះដូង (ជាពិសេស rMSSD) ដើម្បីវាស់ស្ទង់តុល្យភាពរវាងប្រព័ន្ធប្រសាទ sympathetic (ប្រយុទ្ធ-ឬ-ភៀសខ្លួន) និង parasympathetic (សម្រាក-និង-រំលាយអាហារ) របស់អ្នក។

វិទ្យាសាស្ត្រ៖ HRV ខ្ពស់ជាទូទៅត្រូវបានផ្សារភ្ជាប់ទៅនឹងសុខភាពសរសៃឈាមបេះដូងកាន់តែប្រសើរ និងភាពធន់នឹងភាពតានតឹង។ វាបម្រើជាសញ្ញាសម្គាល់ដ៏រសើបសម្រាប់ការរលាកក្នុងប្រព័ន្ធ និងស្ថានភាពនៃការស្តារឡើងវិញ។

ប្រភព៖ Task Force of the European Society of Cardiology (1996); Shaffer & Ginsberg (2017).

ភាពចល័តនៃរាងកាយ (ល្បឿនដើរ)

ល្បឿនដើរជារឿយៗត្រូវបានគេហៅថា \"សញ្ញាសំខាន់ទីប្រាំមួយ\" នៅក្នុងការកំណត់គ្លីនិក។ យើងតាមដាននិន្នាការល្បឿនដើររបស់អ្នក ដើម្បីរកមើលសញ្ញាដំបូងនៃមុខងារចុះខ្សោយ ឬការផ្លាស់ប្តូរប្រព័ន្ធប្រសាទ។

ភាពពាក់ព័ន្ធនឹងគ្លីនិក៖ ការវិភាគលើមនុស្សចាស់ ៣៤,៤៨៥ នាក់បានបង្ហាញថា ល្បឿនដើរត្រូវបានផ្សារភ្ជាប់ដោយផ្ទាល់ទៅនឹងអត្រារស់រានមានជីវិតក្នុងរយៈពេល ៥ ឆ្នាំ និង ១០ ឆ្នាំក្នុងគ្រប់ក្រុមអាយុ។

ប្រភព៖ Studenski S, et al. (JAMA, 2011).

ការគេង និងការសម្អាតកាកសំណល់ក្នុងខួរក្បាល

យើងមិនត្រឹមតែតាមដានម៉ោងគេងប៉ុណ្ណោះទេ យើងវិភាគរយៈពេលនៃការគេងលក់ស្កប់ស្កល់ (NREM ប្រភេទ ៣)។ វាមានសារៈសំខាន់ណាស់សម្រាប់ \"ប្រព័ន្ធ glymphatic\" ដែលជាយន្តការសម្អាតកាកសំណល់របស់ខួរក្បាល។

យន្តការ៖ អំឡុងពេលគេងលក់ស្កប់ស្កល់ ប្រព័ន្ធ glymphatic ក្លាយជាសកម្ម ដោយសម្អាតកាកសំណល់ពុលក្នុងប្រព័ន្ធប្រសាទដូចជា beta-amyloid ដែលកកកុញអំឡុងពេលភ្ញាក់។

ប្រភព៖ Xie L, et al. (Science, 2013); Reddy & van der Werf (2020).

សុខភាពមេតាប៉ូលីស (ការគ្រប់គ្រងជាតិស្ករ)

យើងតាមដានលំនាំជាតិស្ករ និងភាពប្រែប្រួល ដើម្បីវាយតម្លៃសុខភាពមេតាប៉ូលីសលើសពីមធ្យមភាគសាមញ្ញ។ កម្រិតជាតិស្ករដែលមានស្ថិរភាពត្រូវបានផ្សារភ្ជាប់ជាមួយនឹងការកាត់បន្ថយការរលាក និងលទ្ធផលសុខភាពរយៈពេលវែងកាន់តែប្រសើរ។

ចំណុចសំខាន់៖ ភាពប្រែប្រួលនៃជាតិស្ករ (មិនមែនត្រឹមតែកម្រិតមធ្យមភាគទេ) គឺជាកត្តាកំណត់ឯករាជ្យនៃព្រឹត្តិការណ៍សរសៃឈាមបេះដូង និងមរណភាពគ្រប់មូលហេតុ សូម្បីតែចំពោះអ្នកដែលមិនមានជំងឺទឹកនោមផ្អែក៏ដោយ។

ប្រភព៖ Ceriello A, et al. (Diabetes Care, 2008); Hirsch IB (Clinical Diabetes, 2015).

សមាសភាពរាងកាយ (ម៉ាសសាច់ដុំ)

ម៉ាសសាច់ដុំគឺជាជីវសញ្ញាសំខាន់នៃភាពចាស់ប្រកបដោយសុខភាពល្អ។ យើងតាមដាននិន្នាការម៉ាសរាងកាយគ្មានខ្លាញ់ ដើម្បីរកមើលជំងឺ sarcopenia (ការបាត់បង់សាច់ដុំតាមអាយុ) ដែលប៉ះពាល់ដល់កម្លាំង ភាពចល័ត និងសុខភាពមេតាប៉ូលីស។

ចំណងទាក់ទងនឹងភាពយូរអង្វែង៖ ម៉ាសសាច់ដុំខ្ពស់ត្រូវបានផ្សារភ្ជាប់ជាមួយនឹងមរណភាពគ្រប់មូលហេតុទាប។ រាល់ការកើនឡើង ១០% នៃសន្ទស្សន៍ម៉ាសសាច់ដុំ គឺទាក់ទងនឹងការកាត់បន្ថយហានិភ័យមរណភាព ១២%។

ប្រភព៖ Srikanthan P, et al. (American Journal of Medicine, 2014); Cruz-Jentoft AJ, et al. (Age and Ageing, 2019).

ចែករំលែកទិន្នន័យសុខភាពជាមួយគ្រូពេទ្យរបស់អ្នក

រៀបចំសម្រាប់ការណាត់ជួបវេជ្ជសាស្ត្រជាមួយនឹងរបាយការណ៍សុខភាពដ៏ទូលំទូលាយដែលគ្រូពេទ្យរបស់អ្នកអាចប្រើប្រាស់បានជាក់ស្តែង។

ជំនួសឱ្យការព្យាយាមបង្ហាញទិន្នន័យដល់គ្រូពេទ្យនៅលើអេក្រង់ទូរស័ព្ទរបស់អ្នក សូមបង្កើតរបាយការណ៍ PDF ប្រកបដោយវិជ្ជាជីវៈដែលរួមមាន៖

  • រង្វាស់សំខាន់ៗដែលត្រូវបានរំលេច — ផ្តោតលើអ្វីដែលសំខាន់សម្រាប់ការពិគ្រោះយោបល់របស់អ្នក
  • ការមើលឃើញត្រានិន្នាការ — មើលការផ្លាស់ប្តូរតាមពេលវេលាដោយគ្រាន់តែក្រឡេកមើល
  • ចន្លោះយោង — បរិបទសម្រាប់ការបកស្រាយគ្លីនិក
  • ប្រភពដើមនៃទិន្នន័យ — ដឹងថាឧបករណ៍ណាមួយបានកត់ត្រាការវាស់វែងនីមួយៗ

នាំចេញទិន្នន័យពិតប្រាកដដែលអ្នកត្រូវការសម្រាប់រយៈពេលដែលពាក់ព័ន្ធ។ ផ្ញើវាតាមអ៊ីមែលមុនពេលការណាត់ជួបរបស់អ្នក ឬយកច្បាប់ចម្លងដែលបានបោះពុម្ពរួចមកជាមួយ។

ករណីប្រើប្រាស់វេជ្ជសាស្ត្រទូទៅ

  • ការណាត់ជួបផ្នែកបេះដូង — ចង្វាក់បេះដូង, HRV, សម្ពាធឈាម, ទិន្នន័យ ECG
  • ការពិគ្រោះយោបល់អំពីការគេង — ដំណាក់កាលគេង, រយៈពេល, រង្វាស់ការដកដង្ហើម
  • ការគ្រប់គ្រងជំងឺទឹកនោមផ្អែម — ជាតិស្ករក្នុងឈាម, អាំងស៊ុយលីន, ការទទួលទានកាបូអ៊ីដ្រាត
  • ការគ្រប់គ្រងទម្ងន់ — ម៉ាសរាងកាយ, BMI, សកម្មភាព, អាហារូបត្ថម្ភ
  • ការវាយតម្លៃកាយសម្បទា — VO₂ Max, ពេលវេលាហាត់ប្រាណ, រង្វាស់នៃការស្តារឡើងវិញ
